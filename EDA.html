import requests
from xml.etree import ElementTree

def get_exchange_rate(currency_code):
    # URL       
    url = "https://www.cbr.ru/scripts/XML_daily.asp"
    response = requests.get(url)
    response.encoding = 'windows-1251'  #     XML

    #  XML
    tree = ElementTree.fromstring(response.text)

    #      (, USD  EUR)
    for currency in tree.findall("Valute"):
        char_code = currency.find("CharCode").text
        if char_code == currency_code:
            value = currency.find("Value").text.replace(",", ".")
            return float(value)

#   
usd_to_rub = get_exchange_rate("USD")  #    
eur_to_rub = get_exchange_rate("EUR")  #    

print(f"    : {usd_to_rub} .")
print(f"    : {eur_to_rub} .")

def convert_price(n):
    x = n.split()
    if '$' in x[1]:
        return float(x[0])*usd_to_rub
    if '' in n.split()[1]:
        return float(x[0])*eur_to_rub
    else:
        return float(x[0])

df2['price_RUR'] = df2['price'].map(convert_price)

def count_bathrooms(x):
    if isinstance(x, str):
        numbers = re.findall(r'\d+', x)
        return sum(map(int, numbers)) if numbers else np.nan
    return np.nan
